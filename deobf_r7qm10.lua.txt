    Toggles[category] = v

    local dataFolder = Players.LocalPlayer:FindFirstChild("JJKData") or Players.LocalPlayer:FindFirstChild("PlayerData")

        local currentVal = dataFolder:WaitForChild(dataName).Value
        local stop = table.find(Selected[category].Names, dataFolder:WaitForChild(dataName).Value)

            SaveConfig()
            break
        end
        Remotes[remote]:FireServer()
        task.wait(0.6)
    end
end)
    end
end})
